Coding Exercise - FizzBuzz

Enter two numbers between 1 - 100 and submit.
Your results will show the multiples of the first number as "Fizz", and the second number as "Buzz".
If both numbers share the same multiple, it'll result in FizzBuzz!!
Your Results:

                                                            $("#FBclick").click(function () {
                                                                var value1 = Number($("#Fizval").val());
                                                                var value2 = Number($("#Buzval").val());
                                                                var ans = "";

                                                                for (let i = 1; i <= 100; i++) {
                                                                    if (i%value1 == 0 && i%value2 == 0) {
                                                                        ans += ' FizzBuzz ';
                                                                    }
                                                                    else if (i % value1 == 0) {
                                                                        ans += ' Fizz ';
                                                                    }
                                                                    else if (i % value2 == 0) {
                                                                        ans += ' Buzz ';
                                                                    }
                                                                    else {
                                                                        ans += i + ' ';
                                                                    }
                                                                }

                                                                $("#FBres").html(ans);
                                                            });

                                                            $(".clearbtn").click(function () {
                                                                $("#FBres").empty();
                                                                $("#Fizval").val("");
                                                                $("#Buzval").val("");
                                                                $("#FBres").text("");
                                                            });

                                                            $("#Fizval").keypress(function (e) {
                                                                if (!(e.keyCode >= 48 && e.keyCode <= 57) || (e.keyCode >= 96 && e.keyCode <= 105)) {
                                                                    return false;
                                                                }
                                                            });

                                                            $("#Buzval").keypress(function (e) {
                                                                if (!(e.keyCode >= 48 && e.keyCode <= 57) || (e.keyCode >= 96 && e.keyCode <= 105)) {
                                                                    return false;
                                                                }
                                                            });

Coding Exercise - Palindrome

Input a word below:
Your Results:

                                    $("#palclick").click(function () {
                                        var pldm = $("#palinput").val();
                                        var len = pldm.length - 1;
                                        var rev = "";
                                        for (i = len; i >= 0; i--) {
                                            rev += pldm[i];
                                        }
                                        if (pldm == rev) {
                                            $("#palresults").text(pldm + " is a palindrome.");
                                        }
                                        else {
                                            $("#palresults").text(pldm + " is NOT a palindrome.");
                                        }
                                    });
                                    $(".clearbtn").click(function () {
                                        $("#palresults").empty();
                                        $("#palinput").val("");
                                    });

                                    $("#palcode").hide();
                                    $("#paltglOn").click(function () {
                                        $("#palcode").toggle();

                                        if ($(this).text() == "Show Code") {
                                            $(this).text("Hide Code");
                                        }
                                        else {
                                            $(this).text("Show Code")
                                        };
                                    });

                                                                var num1 = Number($("#5input1").val());

                                                                var num4 = Number($("#5input4").val());

                                                                var least = Math.min(num1, num2, num3, num4, num5);

                                                                var sum = num1 + num2 + num3 + num4 + num5;
